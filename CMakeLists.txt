cmake_minimum_required(VERSION 3.0)
project(fixed)

        VERSION 0.0.1
        )



set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_FLAGS "-Wall -Wextra -std=c++14 -O0 -g -pthread")
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

add_executable(fixed
        src/main.cpp
        )

        src/test.cpp
        googletest/googletest/src/gtest-all.cc
        )

        include

        "${CMAKE_CURRENT_SOURCE_DIR}/googletest/googletest/include"
        "${CMAKE_CURRENT_SOURCE_DIR}/googletest/googletest"
        "${CMAKE_CURRENT_SOURCE_DIR}/include"
)

        COMMAND gcovr -b -s -j 4 -r .. --exclude-unreachable-branches -e '.*googletest.*' --html-details --html-title 'Th√©orie des langages' -o html/coverage.html
        )
